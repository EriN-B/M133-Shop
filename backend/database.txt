CREATE TABLE categories (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    categrie_name TEXT
);

CREATE TABLE products (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    product_title TEXT,
    price REAL,
    image TEXT,
    categorie_id INTEGER,
    FOREIGN KEY(categorie_id) REFERENCES categories(id)
);
